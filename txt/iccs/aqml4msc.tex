\subsection{Hybrid Classical-Quantum Neural Network}

\begin{equation}
	\langle \sigma_Z^{(j)} \rangle = \bra{\phi(x)}U^\dagger  \sigma_Z^{(j)}  U \ket{\phi(x)}, \qquad j = 1,2,3
\end{equation}

The measurement results correspond to the expectation values of the $\sigma_Z$ operator acting independently on the first three qubits.
\begin{equation}
	\sigma_Z^{(j)} = \mathbb{I}^{\otimes (j-1)} \otimes \sigma_Z \otimes \mathbb{I}^{\otimes (n-j)}
\end{equation}

State preparation is performed using angle encoding
\begin{equation}
	\ket{\phi(x)} = \left( R_X(x_1) \otimes R_X(x_2) \otimes ... \otimes R_X(x_n) \right) \ket{0}
\end{equation}


The ansatz used in the circuit is constructed from $L$ Basic Entangler Layers provided by the PennyLane package.
\begin{equation}
	U(\boldsymbol{\theta}) = \prod_{\ell=1}^{L} U_{\mathrm{BEL}}(\boldsymbol{\theta}^{(\ell)})
\end{equation}

A single Basic Entangler Layer is defined as:
\begin{equation}
	U_{\mathrm{BEL}}(\boldsymbol{\theta}) =
	\mathrm{CNOT}_{n,1}
	\left(\prod_{i=1}^{n-1} \mathrm{CNOT}_{i,i+1}\right)
	\left(\bigotimes_{i=1}^{n} R_Y(\theta_i)\right)
\end{equation}

Where:
\begin{align*}
	R_Y(\theta) & = e^{-i \theta \sigma_Y / 2}                     \\
	            & = \begin{pmatrix}
		                cos(\frac{\theta}{2}) & -sin(\frac{\theta}{2}) \\
		                sin(\frac{\theta}{2}) & cos(\frac{\theta}{2})
	                \end{pmatrix}
\end{align*}

\begin{equation}
	\mathrm{CNOT}_{i,j}
	=
	\ket{0}\!\bra{0}_i \otimes I_{j}
	+
	\ket{1}\!\bra{1}_i \otimes X_{j}
	, \qquad i \neq j
\end{equation}
